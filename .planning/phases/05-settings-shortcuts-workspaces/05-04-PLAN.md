StorageManager workspace methods — add after clearAllData():
```javascript
/**
 * SESS-03: Persist a WorkspaceSnapshot to storage.local.
 * Enforces MAX_WORKSPACES limit by removing the oldest workspace first.
 * Returns { success: boolean, workspace?: WorkspaceSnapshot, error?: string }.
 */
async function saveWorkspace(snapshot) {
  try {
    const existing = await loadWorkspaces();
    let workspaces = Array.isArray(existing) ? existing : [];

    // Enforce 20-workspace limit — remove oldest if at limit
    const MAX = (typeof CONSTANTS !== 'undefined' && CONSTANTS.MAX_WORKSPACES) || 20;
    if (workspaces.length >= MAX) {
      workspaces = workspaces.slice(1); // oldest is at index 0 (push appends)
    }
    workspaces.push(snapshot);

    await BrowserAdapter.storage.local.set({
      [CONSTANTS.STORAGE_KEYS.WORKSPACES || 'tabnest_workspaces']: workspaces,
    });
    return { success: true, workspace: snapshot };
  } catch (err) {
    console.error('[TabNest] StorageManager.saveWorkspace() failed:', err);
    return { success: false, error: err.message };
  }
}

/**
 * SESS-03: Load all saved WorkspaceSnapshots from storage.local.
 * Returns [] on error or missing key.
 */
async function loadWorkspaces() {
  try {
    const key = (typeof CONSTANTS !== 'undefined' && CONSTANTS.STORAGE_KEYS && CONSTANTS.STORAGE_KEYS.WORKSPACES)
      ? CONSTANTS.STORAGE_KEYS.WORKSPACES
      : 'tabnest_workspaces';
    const result = await BrowserAdapter.storage.local.get(key);
    const wks = result[key];
    return Array.isArray(wks) ? wks : [];
  } catch (err) {
    console.error('[TabNest] StorageManager.loadWorkspaces() failed:', err);
    return [];
  }
}

/**
 * SESS-03: Delete a saved workspace by ID.
 * Returns { success: boolean, error?: string }.
 */
async function deleteWorkspace(workspaceId) {
  try {
    const existing = await loadWorkspaces();
    const filtered = existing.filter(w => w.workspaceId !== workspaceId);
    await BrowserAdapter.storage.local.set({
      [(typeof CONSTANTS !== 'undefined' && CONSTANTS.STORAGE_KEYS && CONSTANTS.STORAGE_KEYS.WORKSPACES)
        ? CONSTANTS.STORAGE_KEYS.WORKSPACES : 'tabnest_workspaces']: filtered,
    });
    return { success: true };
  } catch (err) {
    console.error('[TabNest] StorageManager.deleteWorkspace() failed:', err);
    return { success: false, error: err.message };
  }
}
```

case MSG_TYPES.SAVE_WORKSPACE: {
  const name = (data && data.name) ? String(data.name).trim() : 'Workspace ' + new Date().toLocaleDateString();
  if (!name) return { success: false, error: 'Workspace name cannot be empty' };

  const savedState = globalThis._savedState || { savedEntries: [], groups: [] };

  // Collect simplified tab entries from the live tabRegistry
  const tabEntries = [];
  for (const [, entry] of tabRegistry) {
    if (entry.isInternal || entry.stage === 'discarded') continue;
    tabEntries.push({
      url:     entry.url,
      title:   entry.title,
      favicon: entry.favicon,
      groupId: entry.groupId,
    });
  }

  const snapshot = {
    workspaceId:  (typeof crypto !== 'undefined' && crypto.randomUUID)
                    ? crypto.randomUUID()
                    : Date.now().toString(36) + Math.random().toString(36).slice(2),
    name:         name,
    createdAt:    Date.now(),
    groups:       Array.isArray(savedState.groups) ? savedState.groups.slice() : [],
    tabEntries:   tabEntries,
    savedEntries: Array.isArray(savedState.savedEntries) ? savedState.savedEntries.slice() : [],
  };

  const result = await StorageManager.saveWorkspace(snapshot);
  if (result.success) {
    pushToSidebar(MSG_TYPES.WORKSPACE_SAVED, { workspace: result.workspace });
  }
  return result;
}

case MSG_TYPES.RESTORE_WORKSPACE: {
  const { workspaceId } = data || {};

  const workspaces = await StorageManager.loadWorkspaces();
  const workspace  = workspaces.find(w => w.workspaceId === workspaceId);

    ...(Array.isArray(workspace.savedEntries) ? workspace.savedEntries : []).map(e => e.url),

(function () {

  /**

    section.id        = 'tn-workspaces';
    section.className = 'tn-workspaces-section';
    section.setAttribute('aria-label', 'Saved workspaces');

    countBadge.className   = 'tn-group-count';
    countBadge.textContent = workspaces.length + ' saved';

    toggle.addEventListener('click', function () {
      const expanded = toggle.getAttribute('aria-expanded') === 'true';
      toggle.setAttribute('aria-expanded', String(!expanded));
      list.hidden = expanded;
    });

    header.appendChild(toggle);
    header.appendChild(countBadge);
    section.appendChild(header);

    // Workspace list
    const list = document.createElement('ul');
    list.id        = 'tn-workspace-list';
    list.className = 'tn-workspace-list';
    list.setAttribute('role', 'list');

    if (!workspaces || workspaces.length === 0) {
      const empty = document.createElement('li');
      empty.className   = 'tn-workspace-empty';
      empty.textContent = 'No saved workspaces yet.';
      list.appendChild(empty);
    } else {
      for (var i = 0; i < workspaces.length; i++) {
        list.appendChild(_buildWorkspaceItem(workspaces[i], onRestore, onDelete));
      }
    }

    section.appendChild(list);
    return section;
  }

  function _buildWorkspaceItem(workspace, onRestore, onDelete) {
    const li = document.createElement('li');
    li.className        = 'tn-workspace-item';
    li.setAttribute('role', 'listitem');

    const info = document.createElement('div');
    info.className = 'tn-workspace-info';

    const name = document.createElement('span');
    name.className   = 'tn-workspace-name';
    name.textContent = workspace.name || 'Unnamed Workspace';

    const meta = document.createElement('span');
    meta.className = 'tn-workspace-meta';
    const tabCount  = (workspace.tabEntries   || []).length;
    const savedCount = (workspace.savedEntries || []).length;
    const date = workspace.createdAt ? new Date(workspace.createdAt).toLocaleDateString() : '';
    meta.textContent = tabCount + ' tabs, ' + savedCount + ' saved \u2014 ' + date;

    info.appendChild(name);
    info.appendChild(meta);

    const actions = document.createElement('div');
    actions.className = 'tn-workspace-actions';

    const restoreBtn = document.createElement('button');
    restoreBtn.className   = 'tn-workspace-restore-btn';
    restoreBtn.setAttribute('aria-label', 'Restore workspace: ' + (workspace.name || 'Unnamed'));
    restoreBtn.textContent = 'Restore';
    restoreBtn.addEventListener('click', function () {
      if (typeof onRestore === 'function') onRestore(workspace.workspaceId);
    });

    const deleteBtn = document.createElement('button');
    deleteBtn.className   = 'tn-workspace-delete-btn';
    deleteBtn.setAttribute('aria-label', 'Delete workspace: ' + (workspace.name || 'Unnamed'));
    deleteBtn.textContent = '\u00d7';
    deleteBtn.addEventListener('click', function () {
      if (typeof onDelete === 'function') onDelete(workspace.workspaceId);
    });

    actions.appendChild(restoreBtn);
    actions.appendChild(deleteBtn);

    li.appendChild(info);
    li.appendChild(actions);
    return li;
  }

  /**
   * Build and return an inline save-workspace prompt element.
   * The prompt contains a text input for the workspace name and Confirm/Cancel buttons.
   *
   * @param {function(string):void} onConfirm - Called with the name when user confirms
   * @param {function():void}       onCancel  - Called when user cancels
   * @returns {HTMLElement} <div> prompt element
   */
  function renderSavePrompt(onConfirm, onCancel) {
    const prompt = document.createElement('div');
    prompt.className = 'tn-workspace-prompt';
    prompt.setAttribute('role', 'dialog');
    prompt.setAttribute('aria-label', 'Save workspace');
    prompt.setAttribute('aria-modal', 'true');

    const label = document.createElement('label');
    label.setAttribute('for', 'tn-workspace-name-input');
    label.className   = 'tn-workspace-prompt-label';
    label.textContent = 'Workspace name:';

    const input = document.createElement('input');
    input.type        = 'text';
    input.id          = 'tn-workspace-name-input';
    input.className   = 'tn-workspace-name-input';
    input.placeholder = 'e.g. Monday Morning';
    input.maxLength   = 50;
    input.setAttribute('aria-label', 'Workspace name');

    const btnRow = document.createElement('div');
    btnRow.className = 'tn-workspace-prompt-btns';

    const confirmBtn = document.createElement('button');
    confirmBtn.className   = 'tn-workspace-confirm-btn';
    confirmBtn.textContent = 'Save';
    confirmBtn.setAttribute('aria-label', 'Save workspace');

    const cancelBtn = document.createElement('button');
    cancelBtn.className   = 'tn-workspace-cancel-btn';
    cancelBtn.textContent = 'Cancel';
    cancelBtn.setAttribute('aria-label', 'Cancel save workspace');

    function doConfirm() {
      const name = input.value.trim();
      if (!name) { input.focus(); return; }
      if (typeof onConfirm === 'function') onConfirm(name);
    }

    confirmBtn.addEventListener('click', doConfirm);
    cancelBtn.addEventListener('click', function () {
      if (typeof onCancel === 'function') onCancel();
    });
    input.addEventListener('keydown', function (e) {
      if (e.key === 'Enter')  { e.preventDefault(); doConfirm(); }
      if (e.key === 'Escape') { e.preventDefault(); if (typeof onCancel === 'function') onCancel(); }
    });

    btnRow.appendChild(confirmBtn);
    btnRow.appendChild(cancelBtn);

    prompt.appendChild(label);
    prompt.appendChild(input);
    prompt.appendChild(btnRow);

    // Auto-focus input after insertion (caller must insert into DOM first)
    setTimeout(function () { input.focus(); }, 0);

    return prompt;
  }

  globalThis.WorkspaceManager = { render: render, renderSavePrompt: renderSavePrompt };
})();

  if (!container) return;
  container.innerHTML = '';
  const section = WorkspaceManager.render(
    workspaces,
    function onRestore(workspaceId) {

    },
    function onDelete(workspaceId) {

        function (resp) {

              if (r && r.success) _renderWorkspaces(r.data.workspaces);
            });
          }
        }
      );
    }
  );
  container.appendChild(section);
}
