Create `sidebar/search-bar.js` as an IIFE + globalThis export:

```javascript
/**
 * TabNest Search Bar — sidebar/search-bar.js
 *
 * Real-time search filtering for all lifecycle stages.
 * UI-04: < 100ms for 200 entries.
 *
 * Pattern: IIFE + globalThis.SearchBar export.
 */
(function () {
  'use strict';

  /**
   * Filter tabs, savedEntries, and groups by a search query.
   * Matching is case-insensitive. Matches on title, URL, or group name.
   * Empty query returns everything unchanged (no allocation).
   *
   * @param {string} query
   * @param {TabEntry[]} tabs
   * @param {SavedTabEntry[]} savedEntries
   * @param {TabGroup[]} groups
   * @returns {{ tabs: TabEntry[], savedEntries: SavedTabEntry[], groups: TabGroup[], matchCount: number }}
   */
  function filter(query, tabs, savedEntries, groups) {
    if (!query || !query.trim()) {
      return { tabs, savedEntries, groups, matchCount: tabs.length + savedEntries.length };
    }

    const q = query.trim().toLowerCase();

    // Build a set of group IDs whose name matches the query (for group-name search)
    const matchingGroupIds = new Set();
    for (const g of groups) {
      if (g.name.toLowerCase().includes(q)) matchingGroupIds.add(g.id);
    }

    const filteredTabs = tabs.filter(t =>
      (t.title  && t.title.toLowerCase().includes(q)) ||
      (t.url    && t.url.toLowerCase().includes(q))   ||
      matchingGroupIds.has(t.groupId)
    );

    const filteredSaved = savedEntries.filter(s =>
      (s.title  && s.title.toLowerCase().includes(q)) ||
      (s.url    && s.url.toLowerCase().includes(q))   ||
      matchingGroupIds.has(s.groupId)
    );

    // Only include groups that have at least one matching entry
    const usedGroupIds = new Set([
      ...filteredTabs.map(t => t.groupId),
      ...filteredSaved.map(s => s.groupId),
    ]);
    const filteredGroups = groups.filter(g => usedGroupIds.has(g.id));

    return {
      tabs:         filteredTabs,
      savedEntries: filteredSaved,
      groups:       filteredGroups,
      matchCount:   filteredTabs.length + filteredSaved.length,
    };
  }

  globalThis.SearchBar = { filter };
})();
```

3. Add `updateSearchResultCount()` helper:
```javascript
function updateSearchResultCount(matchCount, totalCount) {
  const el = document.getElementById('tn-search-count');
  if (!el) return;
  if (!_searchQuery) {
    el.textContent = '';
    el.hidden = true;
  } else {
    el.textContent = `${matchCount} of ${totalCount} tabs`;
    el.hidden = false;
  }
}
```

const searchInput = document.getElementById('tn-search');
if (searchInput) {
